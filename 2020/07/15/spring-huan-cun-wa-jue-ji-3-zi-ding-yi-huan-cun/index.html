Spring缓存挖掘机（三）- 自定义缓存

如何实现多级缓存

参考上一篇的源码分析，Spring提供了缓存实现接口，并且又在support包中实现了几种缓存方案，但都仅支持单一缓存。
这时，我们可以把需要的几种缓存组件组装起来，新建一个缓存类Cache 和缓存处理器类CacheManager。
其中Cache类同样继承AbstractValueAdaptingCache，重写其方法，将几种缓存的功能实现封装起来
CacheManager类同样实现CacheManager，重写其方法，在创建缓存实例的地方，改为创建多种缓存实例。

自定义缓存

public class SaasCache<K, V> extends AbstractValueAdaptingCache {

    private Logger log = LoggerFactory.getLogger(SaasCache.class);

    /**
     * 缓存名称
     */
    private final String name;

    /**
     * 缓存级别：1级读本地，2级读redis
     */
    private final Integer cacheLevel;

    private final RedisCache rediscache;
    private final CaffeineCache caffeineCache;

    /**
     *
     * @param name
     * @param cacheLevel
     * @param allowNullValues
     * @param caffeineCache 不传，则表示不存本地
     * @param rediscache    不传，则表示不存redis
     */
    protected SaasCache(String name, Integer cacheLevel, boolean allowNullValues, CaffeineCache caffeineCache, RedisCache rediscache) {
        super(allowNullValues);
        this.name = name;
        this.cacheLevel = cacheLevel;
        if(Objects.isNull(caffeineCache) && Objects.isNull(rediscache)) {
            throw new IllegalArgumentException("at least one cache!");
        }
        if(this.cacheLevel == 2 && Objects.isNull(rediscache)) {
            throw new IllegalArgumentException("redis cache cannot be empty when setting cache level to 2!");
        }
        if(this.cacheLevel == 1 && Objects.isNull(caffeineCache)) {
            throw new IllegalArgumentException("caffeine cache cannot be empty when setting cache level to 1!");
        }
        this.caffeineCache = caffeineCache;
        this.rediscache = rediscache;
    }

    /**
     * spring 内部根据key获取缓存时用到
     * @param key
     * @return
     */
    @Override
    protected Object lookup(Object key) {
        return getIfPresent(key, null);
    }

    @Override
    public String getName() {
        return this.name;
    }

    @Override
    public Object getNativeCache() {
        return this;
    }

    private V getIfPresent(Object key, Callable callable) {
        Object value = null;
        if(cacheLevel == 1) {
            value = getLevel_1(key, callable);
        }
        if(cacheLevel == 2 && Objects.isNull(value)) {
            value = getLevel_2(key, callable);
        }
        if(Objects.nonNull(value)) {
            return (V) value;
        }
        return null;
    }

    @Override
    public <V> V get(Object key, Callable<V> callable) {
        try {
            V v = (V) getIfPresent(key, callable);
            if(Objects.isNull(v)) {
                v = callable.call();
                if(Objects.nonNull(v)) {
                    this.put(key, v);
                }
            }
            return v;
        } catch (Exception e) {
            e.printStackTrace();
        }
        return null;
    }

    @Override
    public ValueWrapper get(Object key) {
        return super.get(key);
    }

    @Override
    public void put(Object key, Object value) {
        log.debug("setting key {}, value {}", key, value);
        if(Objects.isNull(value)) {
            return;
        }
        if(cacheLevel >= 1 && Objects.nonNull(caffeineCache)) {
            caffeineCache.put(key, value);
        }
        if(cacheLevel >= 2) {
            rediscache.put(key, value);
        }
    }

    @Override
    public ValueWrapper putIfAbsent(Object o, Object o1) {
        return null;
    }

    /**
     * 删除指定key，当使用@CacheEvict注解后调用
     * @see org.springframework.cache.interceptor.CacheAspectSupport#processCacheEvicts(java.util.Collection, boolean, java.lang.Object)
     * @param key
     */
    @Override
    public void evict(Object key) {
        caffeineCache.evict(key);
        rediscache.evict(key);
    }

    @Override
    public void clear() {
        caffeineCache.clear();
        rediscache.clear();
    }

    /**
     * 这里获取二级缓存有点区别，当获取不到二级缓存的时候，会尝试去获取一级缓存
     */
    private V getLevel_2(Object key, Callable callable) {
        Object value = null;
        if(Objects.nonNull(callable)) {
            value = rediscache.get(key, callable);
        } else {
            ValueWrapper valueWrapper = rediscache.get(key);
            if(Objects.nonNull(value)) {
                value = valueWrapper.get();
            }
        }
        log.info("query level_2 key: {}, value: {}", key, value);
        if(Objects.isNull(value) && Objects.nonNull(caffeineCache)) {
            value = getLevel_1(key, callable);
        }
        return (V) value;
    }

    private V getLevel_1(Object key, Callable callable) {
        Object value = null;
        if(Objects.nonNull(callable)) {
            value = caffeineCache.get(key, callable);
        } else {
            ValueWrapper valueWrapper = caffeineCache.get(key);
            if(Objects.nonNull(value)) {
                value = valueWrapper.get();
            }
        }
        log.info("query level_1 key: {}, value: {}", key, value);
        if(Objects.nonNull(value)) {
            return (V) value;
        }
        return null;
    }
}

重新实现RedisCache

重新实现的目的是为了使用它的构造函数，RedisCache默认的构造函数是protected的，自定义缓存管理器无法去加载redis缓存实例

public class SaasRedisCache extends RedisCache {
    /**
     * Create new {@link RedisCache}.
     *
     * @param name        must not be {@literal null}.
     * @param cacheWriter must not be {@literal null}.
     * @param cacheConfig must not be {@literal null}.
     */
    protected SaasRedisCache(String name, RedisCacheWriter cacheWriter, RedisCacheConfiguration cacheConfig) {
        super(name, cacheWriter, cacheConfig);
    }
}

自定义缓存管理器

public class SaasCacheManager implements CacheManager {

    private final ConcurrentMap<String, Cache> cacheMap = new ConcurrentHashMap<>(16);

    private RedisConnectionFactory redisConnectionFactory;

    private boolean dynamic = true;

    public Integer cacheExpire;

    public Integer cacheLevel;

    public SaasCacheManager(RedisConnectionFactory redisConnectionFactory, String cacheName, Integer cacheExpire, Integer cacheLevel) {
        this.redisConnectionFactory = redisConnectionFactory;
        this.cacheExpire = cacheExpire;
        this.cacheLevel = cacheLevel;
        setCacheNames(Lists.newArrayList(cacheName));
    }

    public void setCacheNames(@Nullable Collection<String> cacheNames) {
        if (cacheNames != null) {
            for (String name : cacheNames) {
                this.cacheMap.put(name, newCache(name));
            }
            this.dynamic = false;
        }
        else {
            this.dynamic = true;
        }
    }

    /**
     * 本地缓存默认5分钟过期
     * @return
     */
    private CaffeineCache newCaffeineCache(String name) {
        return new CaffeineCache(name, Caffeine.newBuilder().expireAfterWrite(cacheExpire, TimeUnit.MINUTES).expireAfterAccess(cacheExpire, TimeUnit.MINUTES).maximumSize(100).build(), true);
    }

    private RedisCache newRedisCache(String name) {
        RedisCacheWriter redisCacheWriter = RedisCacheWriter.nonLockingRedisCacheWriter(redisConnectionFactory);
        RedisSerializer<Object> jsonSerializer = new GenericFastJsonRedisSerializer();
        RedisSerializationContext.SerializationPair<Object> pair = RedisSerializationContext.SerializationPair.fromSerializer(jsonSerializer);
        RedisCacheConfiguration cacheConfiguration = RedisCacheConfiguration.defaultCacheConfig().serializeValuesWith(pair).entryTtl(Duration.ofMinutes(cacheExpire));
        return new SaasRedisCache(name, redisCacheWriter, cacheConfiguration);
    }

    private Cache newCache(String name) {
        return new SaasCache<String, Object>(name, cacheLevel, true, newCaffeineCache(name), newRedisCache(name));
    }

    @Override
    public Cache getCache(String name) {
        Cache cache = this.cacheMap.get(name);
        if(Objects.isNull(cache) && this.dynamic) {
            synchronized (this.cacheMap) {
                cache = this.cacheMap.get(name);
                if(Objects.isNull(cache)) {
                    cache = newCache(name);
                    this.cacheMap.put(name, cache);
                }
            }
        }
        return cache;
    }

    @Override
    public Collection<String> getCacheNames() {
        return Collections.unmodifiableSet(this.cacheMap.keySet());
    }

}

@EnableCaching
@Configuration
public class SaasCacheConfig {

    /**
     * 全局缓存名称
     */
    public static final String cacheName = "saas-cache";
    /**
     * 缓存过期时间，单位：分钟
     */
    public Integer cacheExpire;
    /**
     * 缓存级别，1：本地caffeine，2：redis
     */
    public Integer cacheLevel;

    @Autowired
    RedisConnectionFactory redisConnectionFactory;

    @Bean(name = {"cacheManager"})
    public CacheManager getCache() {
        return new SaasCacheManager(redisConnectionFactory, cacheName, cacheExpire, cacheLevel);
    }
}

测试方法

@Service
public class DemoServiceImpl implements DemoService {
    @Cacheable(value = SaasCacheConfig.cacheName, key = "#s1.concat(#s2)", sync = true)
    @Override
    public String getData(String s1, String s2, Integer i1) {
        System.out.println("none cache, start to save...");
        return "success";
    }
    @CacheEvict(value = SaasCacheConfig.cacheName, key = "#s1.concat(#s2)", beforeInvocation = false)
    @Override
    public void clearData(String s1, String s2, Integer i1) {
        // empty list
    }
}
